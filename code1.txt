		a[i]=Integer.parseInt(s[i]);
	}

				  temp=a[j];
				  a[j]=a[j+1];
				  a[j+1]=temp;
			  }
		  }
		  }

		  {

			  if(a[j] > a[j+1])
			  {
				  temp=a[j];
				  a[j]=a[j+1];
				  a[j+1]=temp;
			  }
		  }
